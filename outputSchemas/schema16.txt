  sh:targetClass ex:iso8601;
  sh:property [
    sh:path ex:iso8601;
    sh:datatype xsd:string;
    sh:description "e.g. 2014-06-29";
    sh:pattern "^([1-2][0-9]{3}-[0-1][0-9]-[0-3][0-9]|[1-2][0-9]{3}-[0-1][0-9]|[1-2][0-9]{3})$";
  ].
