@prefix xsd: <http://www.w3.org/2001/XMLSchema#> .
@prefix sh: <http://www.w3.org/ns/shacl#> .
@prefix ex: <http://example.org/> .

ex:JS_id_Shape a sh:NodeShape;
  sh:targetClass ex:JS_id;
  sh:property [
    sh:path ex:sdk;
    sh:node ex:sdk_Shape;
  ].

ex:sdk_Shape a sh:NodeShape;
  sh:targetClass ex:sdk;
  sh:property [
    sh:path ex:version;
    sh:datatype xsd:string;
    sh:description "The version of the SDK to use.";
  ];
  sh:property [
    sh:path ex:allowPrerelease;
    sh:datatype xsd:boolean;
  ];
  sh:property [
    sh:path ex:rollForward;
    sh:datatype xsd:string;
    sh:in ("patch,feature,minor,major,latestPatch,latestFeature,latestMinor,latestMajor,disable");
    sh:description "The roll-forward policy when selecting an SDK version, either as a fallback to accommodate missing a specific SDK version or as a directive to use a later version.";
  ].
