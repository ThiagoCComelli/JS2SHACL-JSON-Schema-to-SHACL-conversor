@prefix rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> .
@prefix dash: <http://datashapes.org/dash#> .
@prefix xsd: <http://www.w3.org/2001/XMLSchema#> .
@prefix sh: <http://www.w3.org/ns/shacl#> .
@prefix ex: <http://example.org/> .

ex:coreConfiguration_Shape a sh:NodeShape;
  sh:targetClass ex:coreConfiguration;
  sh:property [
    sh:path ex:language;
    sh:datatype xsd:string;
    sh:description "Describes which programming language the Core is written in";
    sh:in ("csharp");
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:entryPoint;
    sh:datatype xsd:string;
    sh:description "The entry point of the bounded context's Core.  A relative path to the folder";
  ].

ex:interactionLayerConfiguration_Shape a sh:NodeShape;
  sh:targetClass ex:interactionLayerConfiguration;
  sh:property [
    sh:path ex:type;
    sh:datatype xsd:string;
    sh:description "The type of the interaction layer";
    sh:in ("web");
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:language;
    sh:datatype xsd:string;
    sh:description "The programming language of the interaction layer";
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:entryPoint;
    sh:datatype xsd:string;
    sh:description "The entry point of the interaction layer. A relative path to the folder";
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:framework;
    sh:datatype xsd:string;
    sh:description "The framework of the interaction layer";
    sh:in ("react,aurelia");
  ].

ex:resourceTypeImplementationConfiguration_Shape a sh:NodeShape;
  sh:targetClass ex:resourceTypeImplementationConfiguration;
  sh:property [
    sh:path ex:production;
    sh:datatype xsd:string;
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:development;
    sh:datatype xsd:string;
    sh:minCount 1;
  ].

ex:JS_id_Shape a sh:NodeShape;
  sh:targetClass ex:JS_id;
  sh:property [
    sh:path ex:application;
    sh:datatype xsd:string;
    sh:description "The application id that represents the application that this bounded context is a part of";
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:boundedContext;
    sh:datatype xsd:string;
    sh:description "The bounded context id that represents this bounded context";
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:boundedContextName;
    sh:datatype xsd:string;
    sh:description "The bounded context name that represents the name of this bounded context";
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:core;
    sh:node ex:coreConfiguration_Shape;
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:lista;
    sh:node dash:ListShape;
    sh:property [
      sh:path ([sh:zeroOrMorePath rdf:rest] rdf:first);
    ];
  ];
  sh:property [
    sh:path ex:resources;
    sh:node ex:resources_Shape;
    sh:minCount 1;
  ].

ex:resources_Shape a sh:NodeShape;

  sh:property [
    sh:path ex:readModels;
    sh:node ex:resourceTypeImplementationConfiguration_Shape;
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:eventStore;
    sh:node ex:resourceTypeImplementationConfiguration_Shape;
    sh:minCount 1;
  ];
  sh:property [
    sh:path ex:ola;
    sh:node ex:ola_Shape;
  ].

ex:ola_Shape a sh:NodeShape;
  sh:targetClass ex:ola.
